52  /* the ids to cut */
53  double cut[cutsize];
55  if (strcmp(attr, "Random") == 0) {
56  int precut[NODESIZE];
57  /* remove cutsize based on shuffle */
58  for (long int i=0; i<NODESIZE; i++) {
59  precut[i] = i;
60  }
61  shuffle(precut, NODESIZE);
62  for (long int j=0; j<cutsize; j++) {
63  cut[j] = precut[j];
64  }
65  } else {
66  /* check the number of values less than (checkFewer) or equal (checkEqual) to
67  the assigned cutoff value */
68  for (long int i=0; i<NODESIZE; i++) {
69  if (VAN(graph, attr, i) < cutoff) {
70  ++checkFewer; // number of nodes fewer than cutoff
71  } else if (VAN(graph, attr, i) == cutoff){
72  ++checkEqual; // number of nodes equal to cutoff
73  }
74  }
75  for (long int i=0; i<cutsize; i++) {
76  cut[i] = -1.0;
77  }
78  int equal[checkEqual];
79  if (checkFewer == cutsize) {
80  /* if number of equals and less thans are all needed then just do the filter */
81  int index = 0;
82  for (long int i=0; i<NODESIZE; i++) {
83  if (VAN(graph, attr, i) < cutoff) {
84  cut[index] = (double)i;
85  ++index;
86  }
87  }
88  }
89  else if (checkFewer == 0) {
90  /* The only values to cut are equal to the cutoff */
91  printf ("---WARNING--- : Percentage resulted in ambiguous filtering because \n \
92  because no values were lower than cutoff point %f (only equal to) This means\n \
93  that all values at cutoff point will be selected randomly.\n\n", cutoff);
94  int rands = 0;
95  for (long int i=0; i<NODESIZE; i++) {
96  if (VAN(graph, attr, i) == cutoff) {
97  equal[rands] = i;
98  ++rands;
99  }
100  }
101  shuffle(equal, rands);
102  for (long int i=0; i<cutsize; i++) {
103  cut[i] = equal[i];
104  }
105  } else {
106  /* means we have to randomize selection for val == cutoff */
107  int randoms = (cutsize - checkFewer);
108  printf ("---WARNING--- : Percentage resulted in ambiguous filtering.\n This means \
109  that %i values at cutoff point %f \n will be selected randomly.\n\n", randoms, cutoff);
110  int index = 0;
111  int rands = 0;
112  for (long int i=0; i<NODESIZE; i++) {
113  if (VAN(graph, attr, i) < cutoff) {
114  cut[index] = (double)i;
115  ++index;
116  } else if (VAN(graph, attr, i) == cutoff) {
117  equal[rands] = (double)i;
118  ++rands;
119  }
120  }
121  /* second pass add items that equal the cutoff */
122  int ind = 0;
123  int randind = 0;
124  shuffle(equal, rands);
125  for (long int i=0; i<cutsize; i++) {
126  if (cut[ind] == -1) {
127  cut[ind] = (double)equal[randind];
128  ++randind;
129  ++ind;
130  } else {
131  ++ind;
132  }
133  }
134  }
135  }

214 int shrink (igraph_t *graph, int cutsize, char* attr) {
215  igraph_vector_t v;
216  if (strcmp(attr, "Random")==0) {
217  create_filtered_graph(graph, 0.0, cutsize, attr);
218  } else {
219  igraph_vector_init(&v, NODESIZE);
220  for (long i=0; i<NODESIZE; i++) {
221  VECTOR(v)[i] = VAN(graph, attr, i);
222  }

225  }
226  igraph_vector_destroy(&v);
