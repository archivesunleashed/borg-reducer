198 int centralization(igraph_t *graph, char* attr) {
199  long int gsize = (long int)igraph_vcount(graph);
200  int directed = igraph_is_directed(graph) ? 1 : 2;
201  int maximum = (int) (gsize * (gsize -1))/directed;
202  igraph_vector_t scores;
203  char gattr[150];
204  strcpy(gattr, "centralization");
205  strcat(gattr, attr);
206  igraph_vector_init (&scores, NODESIZE);
207  VANV(graph, attr, &scores);
208  igraph_real_t results = 0.0;
209  results = igraph_centralization(&scores, maximum, 1);
210  SETGAN(graph, gattr, results);
211  igraph_vector_destroy(&scores);
212  return 0;
213 }
