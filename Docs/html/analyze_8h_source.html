197 int centralization(igraph_t *graph, char* attr) {
198  long int gsize = (long int)igraph_vcount(graph);
199  int directed = igraph_is_directed(graph) ? 1 : 2;
200  int maximum = (int) (gsize * (gsize -1))/directed;
201  igraph_vector_t scores;
202  char gattr[150];
203  strcpy(gattr, "centralization");
204  strcat(gattr, attr);
205  igraph_vector_init (&scores, NODESIZE);
206  VANV(graph, attr, &scores);
207  igraph_real_t results = 0.0;
208  results = igraph_centralization(&scores, maximum, 1);
209  SETGAN(graph, gattr, results);
210  igraph_vector_destroy(&scores);
211  return 0;
212 }

224 int analysis_all (igraph_t *graph) {
225  calc_authority(graph);
226  calc_betweenness(graph);
227  calc_degree(graph, 'd');
228  calc_hub(graph);
229  calc_degree(graph, 'i');
230  calc_degree(graph, 'o');
231  calc_eigenvector(graph);
232  calc_pagerank (graph);
233  igraph_real_t pathl, cluster, assort, dens, recip;
234  igraph_integer_t dia;
235  igraph_vector_t mod;
236  igraph_vector_init (&mod, igraph_vcount(graph));
237  igraph_average_path_length(graph, &pathl, 1, 1);
238  igraph_diameter(graph, &dia, NULL, NULL, NULL ,1, 1);
239  igraph_transitivity_undirected(graph, &cluster, IGRAPH_TRANSITIVITY_ZERO);
240  calc_modularity(graph);
241  VANV(graph, "WalkTrapModularity", &mod);
242  igraph_assortativity_nominal(graph, &mod, &assort, 1);
243  igraph_density(graph, &dens, 0);
244  igraph_reciprocity(graph, &recip, 1, IGRAPH_RECIPROCITY_DEFAULT);
245  SETGAN(graph, "NODES", igraph_vcount(graph));
246  SETGAN(graph, "EDGES", igraph_ecount(graph));
247  SETGAN(graph, "AVG_PATH_LENGTH", pathl);
248  SETGAN(graph, "DIAMETER", dia);
249  SETGAN(graph, "OVERALL_CLUSTERING", cluster);
250  SETGAN(graph, "ASSORTATIVITY", assort);
251  SETGAN(graph, "DENSITY", dens);
252  SETGAN(graph, "RECIPROCITY", recip);
253  centralization(graph, "Authority");
254  centralization(graph, "Betweenness");
255  centralization(graph, "Degree");
256  centralization(graph, "Hub");
257  centralization(graph, "Indegree");
258  centralization(graph, "Outdegree");
259  centralization(graph, "Eigenvector");
260  centralization(graph, "PageRank");
261  igraph_vector_destroy(&mod);
262  return 0;
263 }
